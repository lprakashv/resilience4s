201   private def handleFailure(block: => R,
202                             throwableOrValue: Either[Throwable, R],
203                             atomicCounter: AtomicInteger,
204                             maxFailures: Int): CircuitResult[R] = {
205     val currentFailureCount = atomicCounter.incrementAndGet()
206     circuitLogger(
207       s"[${state.get()}-error-count = $atomicCounter] $throwableOrValue"
208     )
209     if (currentFailureCount > maxFailures) {
210       openCircuit()
211       execute(block)
212     } else
213       throwableOrValue match {
214         case Left(exception) if ignoreException(exception) =>
215           openCircuit()
216           execute(block)
217         case Left(exception) => CircuitFailure(exception)
218         case Right(value) =>
219           CircuitFailure(InvalidEvalCircuitException(name, value))
220       }
221   }

223   private def handleFailureAsync(block: => Future[R],
224                                  maybeException: Option[Throwable],
225                                  atomicCounter: AtomicInteger,
226                                  maxFailures: Int): Future[CircuitResult[R]] = {
227     val currentFailureCount = atomicCounter.incrementAndGet()
228     circuitLogger(
229       s"[${state.get()}-error-count = $atomicCounter] $maybeException"
230     )
231     if (currentFailureCount > maxFailures) {
232       openCircuit()
233       executeAsync(block)
234     } else
235       maybeException match {
236         case Some(exception) if ignoreException(exception) =>
237           Future.successful(CircuitSuccess(defaultAction))
238         case Some(exception) => Future.successful(CircuitFailure(exception))
239         case None =>
240           Future.successful(
241             CircuitFailure(InvalidEvalCircuitException(name, block))
242           )
243       }
244   }
